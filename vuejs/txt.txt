import java.util.Date;

interface Creatable {
    Date getCreated();
}

class Creature implements Creatable {
    Date d;

    Creature() {
        d = new Date();
    }

    public Date getCreated() {
        return d;
    }
}

class Human extends Creature {
    String n;

    Human(String x) {
        n = x;
    }

    String getName() {
        return n;
    }
}
